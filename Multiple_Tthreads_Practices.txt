Given a list of numbers (large number is better around 952464865).
Compute and print out total divisible number for each element.

Example: 
[ 50, 22 ]

Output:
50 has total 5 divisible numbers
22 has total 3 divisible numbers

using System;
using System.Collections.Generic;
using System.Threading;

namespace C_Sharp_Compiler
{
    class Program
    {
        public static void Main(string[] args)
        {
            Thread divisibilityThr = new Thread(Divisibility);
            divisibilityThr.Start();
        }

        public static void Divisibility ()
        {
            List<int> numbers = new List<int>() { 952464865, 917303614 };

            foreach (int element in numbers)
            {
                int counter = 0;
                for (int i = 1; i <= element; i++)
                {
                    if (element % i == 0)
                        counter++;
                }

                Console.WriteLine($"{element} has {counter} divisible numbers");
            }
        }
    }
}
