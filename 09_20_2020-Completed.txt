1/ Build a book dictionary, and ask a user which book they are looking for, the out put will be the price of that book
Example:
Dictionary:
TitleOfTheBook	Price
Army for you	12
Harry Potter	51
UFO				30

> Which book's title you are looking for?
army

> Army for you cost $12

using System;
using System.Collections.Generic;

class MainClass {
  public static void Main (string[] args) {
    Dictionary <string, double> books = new Dictionary<string, double>()
    {{"food wars", 8.99},
      {"harry potter", 6.59},
      {"v is for vegan", 7.50}
    };
    Console.WriteLine("What book are you looking for? ");
    string input = Console.ReadLine();
    Console.WriteLine("\n{0} costs ${1}", input, books[input.ToLower()]);

  }
}
